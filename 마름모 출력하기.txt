#include <stdio.h>

int main()
{
	int i, j, k, m = 1;		//i:출력할 공백(' ')의 개수, m: 별('*')의 개수

	for (i = 5; i >= 0; i--, m += 2)
	{
		for (j = 0; j < i; j++)
		{
			printf(" ");
		}

		for (k = 0; k < m; k++)
		{
			printf("*");
		}

		printf("\n");
	}

	m -= 4;

	for (i = 1; i <= 5; i++, m -= 2)
	{
		for (j = 0; j < i; j++)
		{
			printf(" ");
		}

		for (k = 0; k < m; k++)
		{
			printf("*");
		}

		printf("\n");
	}

	return 0;
}
