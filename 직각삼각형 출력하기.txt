#include <stdio.h>

void print_star(int line)	//직각삼각형을 출력하는 함수 (line 함수는 정수형인 int형 선언)
{
	int i, j;				//직각삼각형 출력 위치를 지정할 i행, j열을 변수로 선언

	for (i = 0; i < line; i++)		//i행을 0부터 line의 수까지 증가연산자 반복 실행
	{
		for (j = 0; j <= i; j++)	//j열을 0부터 반복문의 i행에 대해 증가된 수만큼 반복 실행
		{
			printf("*");		//출력될 직각삼각형 위치를 '*' 표시를 출력하도록 한다.
		}
		printf("\n");			//한 줄이 끝나면 다음 줄로 개행(New Line)하도록 한다.
	}
}

int main()
{
	int line;					//직각삼각형을 출력할 line 수를 변수에 선언한다.

	printf("몇 줄로 삼각형을 만들 것인가? ");	     //line 수를 지정하기 위한 출력문 
	scanf_s("%d", &line);				//printf() 함수에 의해 입력받을 값을 변수 line에 저장한다.
	print_star(line);				//line 변수에 선언된 값을 이용하여 직각삼각형 출력 함수 선언

	return 0;
}
