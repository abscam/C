//IDE: MS Visual Studio Community 2019

#include <stdio.h>

int main()
{
	int i, j, tmp;			//두 수의 값을 입력 받을 변수(i, j)와 임시변수인 tmp 선언
	int k, sum = 0;			//반복문을 사용하기 위해 쓸 변수(k)와 합계를 저장할 변수(sum) 선언

	printf("시작 숫자를 입력하세요: ");
	scanf_s("%d", &i);		//시작 숫자를 입력받아 변수 i에 저장

	printf("끝 숫자를 입력하세요: ");
	scanf_s("%d", &j);		//끝 숫자를 입력받아 변수 j에 저장

	if (i > j)			//만약 시작숫자가 끝숫자보다 큰 수라면 위치를 바꾸는 조건문 실행
	{
		tmp = i;
		i = j;
		j = tmp;

		printf("\n---[시작숫자가 끝숫자보다 커서 값을 바꾸었습니다.]--- \n");	//if문을 사용했을 경우 값을 바꾸었다는 것을 알려주는 출력문
	}

	printf("\n시작숫자: %d, 끝 숫자: %d \n\n", i, j);	//최종적인 시작 숫자와 끝 숫자를 알려주는 출력문

	for (k = i; k <= j; k++)	//시작 숫자(i)부터 끝 숫자(j)까지 합을 구하는 반복문 실행
	{
		sum = sum + k;
	}

	printf("%d부터 %d까지의 총 합은 %d입니다. \n", i, j, sum);	//결과 출력

	return 0;
}

/*

시작 숫자를 입력하세요: 10
끝 숫자를 입력하세요: 1

---[시작숫자가 끝숫자보다 커서 값을 바꾸었습니다.]---

시작숫자: 1, 끝 숫자: 10

1부터 10까지의 총 합은 55입니다.

*/
