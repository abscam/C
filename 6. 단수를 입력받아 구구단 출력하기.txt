#include <stdio.h>

int main()
{
	int i, j;	//반복문 실행을 위한 변수 i, j 선언

	while (1)	//조건이 없는 불특정 반복문이므로 while() 사용
	{
		printf("출력하고 싶은 단을 입력하세요: ");	
		scanf_s("%d", &i);			//출력하고 싶은 단을 입력받는다. (i)

		for (j = 1; j <= 19; j++)		//출력받고 싶은 단(i)에 대해 19번 곱하는 구구단을 출력하도록 한다.
		{
			printf("%2d * %2d = %3d \n", i, j, i * j);	//구구단 계산 후 결과를 출력하는 출력문
		}
	}
}
