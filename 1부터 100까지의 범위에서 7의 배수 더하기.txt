#include <stdio.h>

int main()
{
	int i;
	int sum = 0;

	puts("[7의 배수] --> [7의 배수의 합] \n");
	for (i = 7; i <= 100; i += 7)
	{
		sum += i;
		printf("[%2d] --> [%3d] \n", i, sum);
	}

	printf("합계: %d \n", sum);

	return 0;
}
