#include <stdio.h>

int main()
{
	int i;		//반복문 실행을 위한 변수 선언
	int sum = 0;	//7의 배수에 대한 합의 값을 저장할 sum 변수 선언(초기값 0 설정)

	puts("[7의 배수] --> [7의 배수의 합] \n");		
	for (i = 7; i <= 100; i += 7)			//7의 배수이기 때문에 7부터 100까지 사이의 7의 배수값에 대한 덧셈을 실행하는 반복문
	{
		sum += i;				//반복문 i가 돌아가면서 7의 배수라고 확인되는 값(i)을 더하여 합계를 저장한다(sum).
		printf("[%2d] --> [%3d] \n", i, sum);	//결과를 출력한다. [%2d, %3d]: 출력할 변수값의 바이트를 지정
	}

	printf("합계: %d \n", sum);			//총 합계에 대한 출력문

	return 0;
}
