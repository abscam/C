#include <stdio.h>

int main()
{
	int i, j, k, m = 1;	//i = 출력할 공백('')의 개수, m = 출력할 별표(*)의 개수

	for (i = 5; i >= 0; i--, m += 2)	//처음 공백(i)개수를 5로 시작하면서 하나씩 줄이고, 별표(m)출력 시작은 1이기 때문에 * 1개로 시작한다.
	{
		for (j = 0; j < i; j++)		//j는 각 줄이다. j가 0부터 i의 초기값만큼 줄을 늘이면서 공백을 출력한다.
		{
			printf(" ");		//공백 출력하는 출력문
		}

		for (k = 0; k < m; k++)		//k는 별표를 출력하는 반복문이다. m의 개수가 m+=2만큼 증가하여 출력됨을 의미한다.
		{
			printf("*");		//별표(*)을 출력한다.
		}
		printf("\n");			//줄바꿈한다. 반복문 종료 때까지의 주기
	}

	m -= 4;					//마지막 별표 4개를 지움으로써 마름모 모양이 바로 잡히게 한다.

	for (i = 1; i <= 5; i++, m -= 2)	//처음 공백(i)개수를 1로 시작하면서 하나씩 늘인다. 위의 반복문에서 생성된 m의 개수만큼의 별표를 2씩 줄여나가서 마름모 모양으로 잡도록 한다.
	{
		for (j = 0; j < i; j++)		//j는 각 줄이다. j가 0부터 i의 초기값만큼 줄을 늘이면서 공백을 출력한다.
		{
			printf(" ");		//공백 출력하는 출력문
		}

		for (k = 0; k < m; k++)		//k는 별표를 출력하는 반복문이다. m의 개수가 m-=2만큼 감소하여 출력됨을 의미한다.
		{
			printf("*");		//별표(*)을 출력한다.
		}
		printf("\n");			//줄바꿈한다. 반복문 종료 때까지의 주기
	}
}
