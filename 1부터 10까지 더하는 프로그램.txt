#include <stdio.h>

int main()
{
	int i;		//반복문 실행을 위한 변수 선언
	int sum = 0;	//합계를 저장할 변수 선언 (0으로 초기화)

	for (i = 1; i <= 10; i++)
	{
		sum = sum + i;			//1부터 10까지 차례대로 더해나가는 반복문 작성
	}

	printf("합은 %d 입니다. \n", sum);		//변수 sum에 선언된 변수값을 이용하여 합계 출력

	return 0;
}

/*

[input]

[output]
합은 55 입니다.

*/
