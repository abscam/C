#include <stdio.h>

int main()
{
	int value;

	while (1)
	{
		printf("숫자를 입력하고 [Enter]를 치세요: ");
		scanf_s("%d", &value);

		if (value % 2 == 0) 
		{
			printf("%d는 짝수입니다. \n", value);
		}
		else
		{
			printf("%d는 홀수입니다. \n", value);
		}
	}
}
